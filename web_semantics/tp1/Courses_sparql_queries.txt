SELECT ?c
WHERE {
	?c :keyword "RDF" .
}

--------------------------------------------------------------------------------

SELECT DISTINCT ?c
WHERE {
	?c :keyword ?k .
	FILTER ( ?k != "RDF" )
}

--------------------------------------------------------------------------------

SELECT DISTINCT ?c
WHERE {
   ?c a :Course .   
   FILTER NOT EXISTS {
        ?c :keyword "RDF" }
}

--------------------------------------------------------------------------------

SELECT DISTINCT ?c
WHERE {
   ?c a :Course .   
   MINUS {
        ?c :keyword "RDF" }
}

SELECT ?m
WHERE {
   ?m a :MasterProgram .

   ?x :course ?c .
   ?c :exam ?e .

   ?m a :MasterProgram .
